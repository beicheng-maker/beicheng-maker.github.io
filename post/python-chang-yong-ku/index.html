import threading
count=0
lock=threading.Lock()
def doAdd():
	global count,lock
	lock.acquire()
	for i in xrange(100):
		count+=1
	lock.release()
for i in range(5):
	threading.Thread(target=doAdd,args=(),name='thread-'+str(i)).start()
